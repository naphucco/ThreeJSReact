# Tạo project mới từ template
npm create vite@latest my-threejs-app -- --template react
cd my-threejs-app

# Cài Three.js và các dependency cần thiết
npm install three @react-three/fiber @react-three/drei
npm install

# Chạy thử local
npm run dev
